🚀 Setup in 3 simple steps

          

            

              
1

              

                
Create a new Google Sheet

                
Open sheets.google.com and create a new sheet. Copy the ID from the URL (the string between /d/ and /edit)

              

            

            

              
2

              

                
Add Apps Script

                
In the sheet: Extensions menu → Apps Script. Delete everything and paste the code below:

                

                  
const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const { action, sheet: sheetName, data } = body;
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    
    if (action === 'write') {
      let sh = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName);
      sh.clearContents();
      if (data && data.length > 0) {
        sh.getRange(1, 1, data.length, data[0].length).setValues(data);
      }
      return ContentService.createTextOutput(JSON.stringify({ok:true}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    if (action === 'read') {
      let sh = ss.getSheetByName(sheetName);
      if (!sh) return ContentService.createTextOutput(JSON.stringify({ok:true,data:[]}))
        .setMimeType(ContentService.MimeType.JSON);
      const vals = sh.getDataRange().getValues();
      return ContentService.createTextOutput(JSON.stringify({ok:true,data:vals}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:'unknown action'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({ok:true,status:'ServicePro API ready'}))
    .setMimeType(ContentService.MimeType.JSON);
}
